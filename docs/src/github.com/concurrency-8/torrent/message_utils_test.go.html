    31  func getRandomByteArr(size uint) []byte {
    32  	temp := make([]byte, size)

    34  	if err != nil {
    35  		fmt.Fprintf(os.Stderr, "Unable to generate Crypto random byte array")

   341  func TestParseMsg(t *testing.T) {
   342  
   343  	// BuildChoke
   344  	chokeMessage, _ := BuildChoke()
   345  	size, id, payload := ParseMsg(chokeMessage)
   346  	assert.Equal(t, size, uint32(1), "choke : Size not equal")
   347  	assert.Equal(t, id, uint8(0), "choke : Message ID different")
   348  	assert.Equal(t, len(payload), 0, "choke : length of payload not zero")
   349  
   350  	//BuildUnchoke
   351  	unchokeMessage, _ := BuildUnchoke()
   352  	size, id, payload = ParseMsg(unchokeMessage)
   353  	assert.Equal(t, size, uint32(1), "unchoke : Size not equal")
   354  	assert.Equal(t, id, uint8(1), "unchoke : Message ID different")
   355  	assert.Equal(t, len(payload), 0, "unchoke : length of payload not zero")
   356  
   357  	// BuildInterested
   358  	interestedMessage, _ := BuildInterested()
   359  	size, id, payload = ParseMsg(interestedMessage)
   360  	assert.Equal(t, size, uint32(1), "Interested : Size not equal")
   361  	assert.Equal(t, id, uint8(2), "Interested : Message ID different")
   362  	assert.Equal(t, len(payload), 0, "Interested: length of payload not zero")
   363  
   364  	// BuildUninerested
   365  	uninterestedMessage, _ := BuildUninterested()
   366  	size, id, payload = ParseMsg(uninterestedMessage)
   367  	assert.Equal(t, size, uint32(1), "UnInterested: Size not equal")
   368  	assert.Equal(t, id, uint8(3), "Unterested: Message ID different")
   369  	assert.Equal(t, len(payload), 0, "UnInterested: length of payload not zero")
   370  
   371  	// BuildHave
   372  	p1 := rand.Uint32()
   373  	haveMessage, _ := BuildHave(p1)
   374  	size, id, payload = ParseMsg(haveMessage)
   375  	assert.Equal(t, size, uint32(5), "Have: Size not equal")
   376  	assert.Equal(t, id, uint8(4), "Have: Message ID different")
   377  	var p2 uint32
   378  	binary.Read(payload["payload"].(*bytes.Buffer), binary.BigEndian, &p2)
   379  	assert.Equal(t, p2, p1, "Have: length of payload not zero")
   380  
   381  	file, _ := parser.ParseFromFile(parser.GetTorrentFileList()[0])
   382  	pieceBlock := parser.RandomPieceBlock(file)
   383  
   384  	// BuildRequest
   385  	requestMessage, _ := BuildRequest(pieceBlock)
   386  	size, id, payload = ParseMsg(requestMessage)
   387  	assert.Equal(t, size, uint32(13), "Request: Size not equal")
   388  	assert.Equal(t, id, uint8(6), "Request: Message ID different")
   389  	assert.Equal(t, uint32(payload["index"].(uint32)), pieceBlock.Index, "Request: index field of payload not same")
   390  	assert.Equal(t, uint32(payload["begin"].(uint32)), pieceBlock.Begin, "Request: begin field of payload not same")
   391  	var length uint32
   392  	binary.Read(payload["length"].(*bytes.Buffer), binary.BigEndian, &length)
   393  	assert.Equal(t, length, pieceBlock.Length, "Request: length field of payload not same")
   394  
   395  	// BuildCancel
   396  	cancelMessage, _ := BuildCancel(pieceBlock)
   397  	size, id, payload = ParseMsg(cancelMessage)
   398  	assert.Equal(t, size, uint32(13), "Cancel: Size not equal")
   399  	assert.Equal(t, id, uint8(8), "Cancel: Message ID different")

   401  	assert.Equal(t, uint32(payload["begin"].(uint32)), pieceBlock.Begin, "Cancel: begin field of payload not same")
   402  	binary.Read(payload["length"].(*bytes.Buffer), binary.BigEndian, &length)
