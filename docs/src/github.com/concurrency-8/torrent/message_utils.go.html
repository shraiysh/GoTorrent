    16  // BuildHandshake returns a pointer to a buffer.
    17  // Buffer looks like:
    18  //	uint8		: pstrlen	- Length of pstr
    19  //	[pstrlen]byte	: pstr		- pstr, the string identifier of the protocol
    20  //	[8]byte		: reserved	- 8 reserved bytes
    21  //	[20]byte	: infohash	- SHA1 hash of the info key in the metainfo file. Same as the info hash transmitted in tracker requests
    22  //	[20]byte	: peerID	- 20 byte unique ID for the client. Usually the same peerID transmitted in tracker requests
    23  // In version 1.0 of the BitTorrent protocol, pstrlen = 19, and pstr = "BitTorrent protocol"
    24  func BuildHandshake(report tracker.ClientStatusReport) (handshake *bytes.Buffer, err error) {
    25  	handshake = new(bytes.Buffer)
    26  
    27  	// pstrlen
    28  	if err = binary.Write(handshake, binary.BigEndian, uint8(19)); err != nil {
    29  		return
    30  	}
    31  
    32  	// pstr
    33  	if err = binary.Write(handshake, binary.BigEndian, []byte("BitTorrent protocol")); err != nil {
    34  		return
    35  	}
    36  
    37  	// reserved
    38  	if err = binary.Write(handshake, binary.BigEndian, uint64(0)); err != nil {
    39  		return
    40  	}
    41  
    42  	// infohash
    43  	var infohashFromFile [20]byte
    44  	copy(infohashFromFile[:], report.TorrentFile.InfoHash)
    45  	if err = binary.Write(handshake, binary.BigEndian, infohashFromFile); err != nil {
    46  		return
    47  	}
    48  
    49  	// peerID
    50  	if err = binary.Write(handshake, binary.BigEndian, []byte(report.PeerID)); err != nil {
    51  		return
    52  	}
    53  
    54  	return
    55  }

   288  // ParseMsg parses a message
   289  func ParseMsg(msg *bytes.Buffer) (size uint32, id uint8, payload Payload) {
   290  	payload = make(Payload)
   291  	binary.Read(msg, binary.BigEndian, &size)
   292  	// fmt.Println("message:", msg.Bytes())
   293  	if size > 0 {
   294  		binary.Read(msg, binary.BigEndian, &id)
   295  	}
   296  	if size > 1 {
   297  
   298  		if id == 6 || id == 7 || id == 8 {
   299  			rest := bytes.NewBuffer(msg.Bytes()[8:])
   300  			var index, begin uint32
   301  			binary.Read(msg, binary.BigEndian, &index)
   302  			binary.Read(msg, binary.BigEndian, &begin)
   303  			payload["index"] = index
   304  			payload["begin"] = begin
   305  
   306  			if id == 7 {
   307  				payload["block"] = rest
   308  				// fmt.Println(rest.Bytes()[:10])
   309  			} else {
   310  				payload["length"] = rest
   311  			}
   312  		} else {
   313  			payload["payload"] = msg
   314  		}
   315  	}
   316  
   317  	return
