   135  func sendHandshake(peer tracker.Peer, report *tracker.ClientStatusReport) (conn net.Conn, err error) {
   136  	buffer, err := BuildHandshake(*report)
   137  	if err != nil {
   138  		return nil, err
   139  	}
   140  	peerip := make([]byte, 4)
   141  	binary.BigEndian.PutUint32(peerip, peer.IPAdress)
   142  	service := net.TCPAddr{
   143  		IP:   peerip,
   144  		Port: int(peer.Port),
   145  		Zone: "",
   146  	}
   147  	Info.Println("peer: <", peer, ">: Dialing TCP connection")
   148  	d := net.Dialer{Timeout: TCPTimeout * time.Second}
   149  	err = nil
   150  	count := 0
   151  	for count < MaxTry {
   152  		count++
   153  		conn, err = d.Dial("tcp", service.String())
   154  		if err != nil {
   155  			Info.Println("peer: <", peer, ">: Unable to set up TCP connection: ", count)
   156  		} else {
   157  			Info.Println("peer: <", peer, ">: Successfully connected to Peer")
   158  			break
   159  		}
   160  	}
   161  
   162  	if err != nil {
   163  		Error.Println("peer: <", peer, ">: Could not connect to peer!")
   164  		return nil, err
   165  	}
   166  	Info.Println("peer: <", peer, ">: Handshaking")
   167  
   168  	//write the handshake content into the connection.
   169  	_, err = conn.Write(buffer.Bytes())
   170  	if err != nil {
   171  		return nil, err
   172  	}
   173  
   174  	return conn, nil

   177  func msgHandler(peer tracker.Peer, msg []byte, conn net.Conn, pieces *piece.PieceTracker, queue *queue.Queue, report *tracker.ClientStatusReport) error {

   224  func onWholeMessage(peer tracker.Peer, conn net.Conn, msgHandler handler, pieces *piece.PieceTracker, queue *queue.Queue, report *tracker.ClientStatusReport) (status int, err error) {
   225  	buffer := new(bytes.Buffer)
   226  	handshake := true
   227  	resp := make([]byte, 1000)
   228  	msgLen := -1
   229  	count := 0
   230  	for pieces != nil && !pieces.IsDone() {
   231  		conn.SetReadDeadline(time.Now().Add(ReadTimeout * time.Second)) // Setting Read deadline from a connection
   232  		respLen, err := conn.Read(resp)
   233  		//Please look for a better connection handling in the future.
   234  		//Maybe use defer?
   235  
   236  		if err != nil {
   237  			if netErr, ok := err.(net.Error); ok && netErr.Timeout() && !handshake {
   238  				count++
   239  				if count < MaxTry {
   240  					Info.Println("Timeout error - Try again")
   241  					continue
   242  				} else {
   243  					Info.Println("peer: <", peer, ">: Many timeout errors - Peer not responding. Should try to reconnect")
   244  					conn.Close()
   245  					return 1, err
   246  				}
   247  			} else if err != io.EOF {
   248  				Error.Println("peer: <", peer, ">: Error while reading from connection: ", err)
   249  				Info.Println("peer: <", peer, ">: Restarting connection")
   250  				conn.Close()
   251  				return 1, err
   252  			} else {
   253  				Error.Println("peer: <", peer, ">: Peer does not respond. Stop contacting this peer.")
   254  				return 0, err
   255  			}
   256  		}
   257  
   258  		binary.Write(buffer, binary.BigEndian, resp[:respLen])
   259  
   260  		if handshake {
   261  			Info.Println("peer: <", peer, ">: First message from peer afte connection starts - Must be handshake")
   262  			length := uint8((buffer.Bytes())[0])
   263  			msgLen = int(length) + 49
   264  		} else if msgLen == -1 {
   265  			length := binary.BigEndian.Uint32(buffer.Bytes()[0:4])
   266  			// length := uint32((buffer.Bytes())[0:4])
   267  			msgLen = int(length) + 4
   268  			Info.Println("peer: <", peer, ">: New message reception started, len =", msgLen)
   269  			// Info.Println("peer: <", peer, ">: Setting msgLen to", msgLen)
   270  		}
   271  
   272  		for len(buffer.Bytes()) >= 4 && msgLen != -1 && len(buffer.Bytes()) >= msgLen {
   273  			Info.Println("peer: <", peer, ">: Message received, msgLen =", msgLen)
   274  			messageBytes := make([]byte, msgLen)
   275  			binary.Read(buffer, binary.BigEndian, messageBytes)
   276  			// Info.Println("peer: <", peer, ">: msgLen:", msgLen)
   277  			msgHandler(peer, messageBytes, conn, pieces, queue, report)
   278  			Info.Println("peer: <", peer, ">: Message handled - setting msgLen = -1")
   279  			msgLen = -1
   280  			handshake = false
   281  			if len(buffer.Bytes()) > 4 {
   282  				length := binary.BigEndian.Uint32(buffer.Bytes()[0:4])
   283  				msgLen = int(length) + 4
   284  				Info.Println("peer: <", peer, ">: New message was in previous one - msgLen =", msgLen)
   285  				// Info.Println("peer: <", peer, ">: Setting msgLen to", msgLen)
   286  			}
   287  		}
   288  	}
   289  	return 0, nil
   290  }

   347  // BitFieldHandler handles bitfield protocol
   348  func BitFieldHandler(peer tracker.Peer, conn net.Conn, pieces *piece.PieceTracker, queue *queue.Queue, payload Payload) (err error) {
   349  	queueempty := (queue.Length() == 0)
   350  	msg := payload["payload"]
   351  	for i, bytevalue := range msg.(*bytes.Buffer).Bytes() {
   352  		for j := 7; j >= 0; j-- {
   353  			if 1 == bytevalue&1 {
   354  				err = queue.Enqueue(uint32(i*8 + j))
   355  			}
   356  			bytevalue = bytevalue >> 1
   357  		}
   358  	}
   359  	if queueempty {
   360  		Info.Println("peer: <", peer, ">: BitFieldHandler: Queue was empty. Requesting pieces")
   361  		err = RequestPiece(peer, conn, pieces, queue)
   362  	}
   363  
   364  	return
   365  }

   371  	Info.Println("peer: <", peer, ">: Received piece[", pieceResp.Index, "] [", pieceResp.Begin/parser.BLOCK_LEN, "]")
   372  	report.Data[pieceResp.Index].Blocks[pieceResp.Begin/parser.BLOCK_LEN] = pieceResp
   373  
   374  	toSHA1 := func(data []byte) []byte {
   375  		hash := sha1.New()

   377  		return hash.Sum(nil)
   378  	}
   379  	var piece []byte
   380  	if pieces.PieceIsDone(pieceResp.Index) {
   381  		for _, i := range report.Data[pieceResp.Index].Blocks {
   382  			if len(i.Bytes) != 0 {

   392  		actual := toSHA1(piece)

   394  			same = same && expected[i] == actual[i]

   445  // RequestPiece requests a piece
   446  func RequestPiece(peer tracker.Peer, conn net.Conn, pieces *piece.PieceTracker, queue *queue.Queue) (err error) {
   447  	if queue.Choked {
   448  		Error.Println("peer: <", peer, ">: Queue is choked")
   449  		return
   450  	}
   451  
   452  	for queue.Length() > 0 {
   453  		pieceBlock, err := queue.Peek()
   454  
   455  		if err != nil {
   456  			break
   457  		}
   458  
   459  		err = queue.Dequeue()
   460  
   461  		if err != nil {
   462  			break
   463  		}
   464  
   465  		pieceTrackerLock.Lock()
   466  		if pieces.Needed(pieceBlock) {
   467  			pieces.AddRequested(pieceBlock)
   468  			pieceTrackerLock.Unlock()
   469  			Info.Println("peer: <", peer, ">: Requesting piece[", pieceBlock.Index, "][", pieceBlock.Begin/parser.BLOCK_LEN, "]")
   470  			message, err := BuildRequest(pieceBlock)
   471  
   472  			if err != nil {
   473  				break
   474  			}
   475  			_, err = conn.Write(message.Bytes())
   476  
   477  			if err != nil {
   478  				Info.Println(err.Error())
   479  				break
   480  			}
   481  			break
   482  		} else {
   483  			pieceTrackerLock.Unlock()
   484  		}
   485  	}
   486  	return
   487  }
